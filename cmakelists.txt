cmake_minimum_required(VERSION 3.12)
project(SimpleRenderer CXX)

# 设置 C++20 标准
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 添加可执行文件
add_executable(SimpleRenderer
    main.cpp
    renderer.cpp
    # maths.cpp
    shader.cpp
)

# 添加头文件目录
target_include_directories(SimpleRenderer PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
